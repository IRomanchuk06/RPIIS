TEST(SumTreeTest, ValidInput) {
    // Создаем объект SumTree и проверяем корректность работы на правильных данных
    std::vector<int> arr = { 1, 2, 3, 4, 5 };
    SumTree tree(arr);

    // Проверка суммы значений на отрезке [2, 4] (ожидаемое значение: 9)
    EXPECT_EQ(tree.query(2, 4), 9);

    // Изменение значений на отрезке [3, 5] на 2
    tree.modify(3, 5, 2);

    // Проверка суммы значений на отрезке [2, 4] после изменения (ожидаемое значение: 15)
    EXPECT_EQ(tree.query(2, 4), 15);
}
